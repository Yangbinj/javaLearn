title Command Pattern Diagram

abstract Command{
    receiver:Receiver
    +execute():void
}
class ConcreCommandA
class ConcreCommandB

class Invoker{
     - command:Command;
     + Invoker(Command command):Invoker
     + request()
}

class Receiver{
    + action();
}

Command <|-- ConcreCommandA
